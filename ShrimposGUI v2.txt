local spinConn
local function startSpin()
    spinConn = RunService.Heartbeat:Connect(function()
        local hrp = getHRP()
        if hrp then
            local speed = _G.SpinSpeed or 50
            hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.rad(speed) * (1/60), 0)
        end
    end)
end
local function stopSpin()
    if spinConn then spinConn:Disconnect(); spinConn = nil end
end

-- =====================
-- T-POSE / MOONWALK
-- =====================
local function applyTPose(on)
    local c = getChar()
    if not c then return end
    local hum = c:FindFirstChildOfClass("Humanoid")
    if not hum then return end
    if on then
        hum.AutoRotate = false
        tposeConn = RunService.Heartbeat:Connect(function()
            local ch = getChar()
            if not ch then return end
            for _, m in ipairs(ch:GetDescendants()) do
                if m:IsA("Motor6D") then
                    m.DesiredAngle = 0
                end
            end
        end)
    else
        if tposeConn then tposeConn:Disconnect(); tposeConn = nil end
        if hum then hum.AutoRotate = true end
    end
end

local function applyFullbright(on)
    if on then
        Lighting.Ambient        = Color3.new(1, 1, 1)
        Lighting.Brightness     = 2
        Lighting.OutdoorAmbient = Color3.new(1, 1, 1)
        for _, v in ipairs(Lighting:GetChildren()) do
            if v:IsA("BlurEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("SunRaysEffect") then
                v.Enabled = false
            end
        end
    else
        Lighting.Ambient        = originalAmbient
        Lighting.Brightness     = originalBrightness
        Lighting.OutdoorAmbient = originalOutdoorAmbient
        for _, v in ipairs(Lighting:GetChildren()) do
            if v:IsA("BlurEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("SunRaysEffect") then
                v.Enabled = true
            end
        end
    end
end

            notify("Cooldown", "Wait 5 seconds before teleporting again!")
            return
        end
        local target
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Name:lower():find(targetPlayerName:lower(), 1, true) then
                target = p; break
            end
        end
        if target and target.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local mHRP = getHRP()
            if tHRP and mHRP then
                mHRP.CFrame = tHRP.CFrame + Vector3.new(3, 0, 0)
                teleportCooldown = true
                notify("Teleported!", "Jumped to " .. target.Name)
                task.delay(5, function() teleportCooldown = false end)
                return
            end
        end
        notify("Error", "Player '" .. targetPlayerName .. "' not found.")
    end,
})

local function applyESP(model, twistedName, rarity)
    local greenTwisteds  = { BobetteMonster=true, RudieMonster=true, GingerMonster=true, CoalMonster=true }
    local purpleTwisteds = { BassieMonster=true, CocoaMonster=true, EggsonMonster=true, FlyteMonster=true }
    local orangeTwisteds = { GourdyMonster=true, EclipseMonster=true, RibbecaMonster=true, SoulvesterMonster=true }
    local blackTwisteds  = { DandyMonster=true, DyleMonster=true }

    local fillColor, outlineColor

    if blackTwisteds[twistedName] then
        fillColor    = Color3.fromRGB(10,  10,  10)
        outlineColor = Color3.fromRGB(0,   0,   0)
    elseif greenTwisteds[twistedName] then
        fillColor    = Color3.fromRGB(0,   180, 0)
        outlineColor = Color3.fromRGB(0,   255, 80)
    elseif purpleTwisteds[twistedName] then
        fillColor    = Color3.fromRGB(120, 0,   200)
        outlineColor = Color3.fromRGB(180, 80,  255)
    elseif orangeTwisteds[twistedName] then
        fillColor    = Color3.fromRGB(220, 100, 0)
        outlineColor = Color3.fromRGB(255, 160, 50)
    else
        fillColor    = Color3.fromRGB(180, 0,   0)
        outlineColor = Color3.fromRGB(255, 50,  50)
    end

    local hl = Instance.new("Highlight")
    hl.FillColor           = fillColor
    hl.OutlineColor        = outlineColor
    hl.FillTransparency    = 0.4
    hl.OutlineTransparency = 0
    hl.Adornee             = model
    hl.Parent              = model
    table.insert(espObjects, hl)

    local rootPart = model:IsA("Model") and (model.PrimaryPart or model:FindFirstChildWhichIsA("BasePart")) or model
    if not rootPart then return end

    local bb = Instance.new("BillboardGui")
    bb.Adornee     = rootPart
    bb.AlwaysOnTop = true
    bb.Size        = UDim2.new(0, 180, 0, 30)
    bb.StudsOffset = Vector3.new(0, 5, 0)
    bb.Parent      = rootPart

    local displayName = twistedName:gsub("Monster", "")

    local lbl = Instance.new("TextLabel")
    lbl.Size                   = UDim2.new(1, 0, 1, 0)
    lbl.BackgroundTransparency = 1
    lbl.Text                   = "Twisted " .. displayName .. " | " .. rarity
    lbl.TextColor3             = rarityTextColors[rarity] or Color3.new(1,1,1)
    lbl.TextStrokeTransparency = 0
    lbl.TextStrokeColor3       = Color3.new(0, 0, 0)
    lbl.Font                   = Enum.Font.GothamBold
    lbl.TextSize               = 11
    lbl.TextScaled             = false
    lbl.Parent                 = bb

    table.insert(espObjects, bb)
end

local function clearPlayerESP()
    for _, obj in ipairs(playerESPObjects) do
        if obj and obj.Parent then obj:Destroy() end
    end
    playerESPObjects = {}
end

local function getPlayerHearts(player)
    local char = player.Character
    if not char then return 0 end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return 0 end
    local hp = math.floor(hum.Health)
    -- Dandy's World: each heart = 100hp, max 4 hearts = 400hp
    if hp >= 400 then return 4
    elseif hp >= 300 then return 3
    elseif hp >= 200 then return 2
    elseif hp >= 100 then return 1
    else return 0
    end
end

local function getHeartColor(hearts, hp)
    if hp >= 400 then
        -- 4 hearts: blue
        return Color3.fromRGB(0, 100, 255), Color3.fromRGB(100, 180, 255)
    elseif hp >= 300 then
        -- 3 hearts: blue
        return Color3.fromRGB(0, 100, 255), Color3.fromRGB(100, 180, 255)
    elseif hp >= 200 then
        -- 2 hearts: dark blue
        return Color3.fromRGB(0, 0, 160), Color3.fromRGB(0, 50, 200)
    elseif hp >= 100 then
        -- 1 heart: blue
        return Color3.fromRGB(0, 100, 255), Color3.fromRGB(100, 180, 255)
    else
        -- 0 hp / dead: black
        return Color3.fromRGB(10, 10, 10), Color3.fromRGB(50, 50, 50)
    end
end

local function isInLobby()
    -- Dandy's World lobby detection: no "Rounds" folder or game not started
    local roundsFolder = Workspace:FindFirstChild("Rounds") or Workspace:FindFirstChild("Round") or Workspace:FindFirstChild("GameFolder")
    local floorFolder  = Workspace:FindFirstChild("Floor") or Workspace:FindFirstChild("Floors")
    return not roundsFolder and not floorFolder
end

local function applyPlayerESP(player)
    local char = player.Character
    if not char or player == LocalPlayer then return end

    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local lobby = isInLobby()

    -- Default blue for lobby or unknown
    local fillColor    = Color3.fromRGB(0, 100, 255)
    local outlineColor = Color3.fromRGB(100, 180, 255)

    if not lobby then
        local hum = char:FindFirstChildOfClass("Humanoid")
        local hp  = hum and math.floor(hum.Health) or 0
        local hearts = getPlayerHearts(player)
        fillColor, outlineColor = getHeartColor(hearts, hp)
    end

    -- Highlight
    local hl = Instance.new("Highlight")
    hl.FillColor           = fillColor
    hl.OutlineColor        = outlineColor
    hl.FillTransparency    = 0.4
    hl.OutlineTransparency = 0
    hl.Adornee             = char
    hl.Parent              = char
    table.insert(playerESPObjects, hl)

    -- BillboardGui label
    local bb = Instance.new("BillboardGui")
    bb.Adornee     = hrp
    bb.AlwaysOnTop = true
    bb.Size        = UDim2.new(0, 220, 0, 70)
    bb.StudsOffset = Vector3.new(0, 3.5, 0)
    bb.Parent      = hrp

    local frame = Instance.new("Frame")
    frame.Size                   = UDim2.new(1, 0, 1, 0)
    frame.BackgroundTransparency = 1
    frame.Parent                 = bb

    if lobby then
        -- Lobby: show username + display name, no hearts
        local topLabel = Instance.new("TextLabel")
        topLabel.Size                   = UDim2.new(1, 0, 0.5, 0)
        topLabel.Position               = UDim2.new(0, 0, 0, 0)
        topLabel.BackgroundTransparency = 1
        topLabel.Text                   = player.Name
        topLabel.TextColor3             = Color3.fromRGB(100, 180, 255)
        topLabel.TextStrokeTransparency = 0
        topLabel.TextStrokeColor3       = Color3.new(0, 0, 0)
        topLabel.Font                   = Enum.Font.GothamBold
        topLabel.TextScaled             = true
        topLabel.Parent                 = frame

        local botLabel = Instance.new("TextLabel")
        botLabel.Size                   = UDim2.new(1, 0, 0.5, 0)
        botLabel.Position               = UDim2.new(0, 0, 0.5, 0)
        botLabel.BackgroundTransparency = 1
        botLabel.Text                   = player.DisplayName
        botLabel.TextColor3             = Color3.new(1, 1, 1)
        botLabel.TextStrokeTransparency = 0
        botLabel.TextStrokeColor3       = Color3.new(0, 0, 0)
        botLabel.Font                   = Enum.Font.Gotham
        botLabel.TextScaled             = true
        botLabel.Parent                 = frame
    else
        -- In game: show name + hearts + HP
        local hum = char:FindFirstChildOfClass("Humanoid")
        local hp  = hum and math.floor(hum.Health) or 0
        local hearts = getPlayerHearts(player)
        local fillColor, outlineColor = getHeartColor(hearts, hp)

        -- Re-apply highlight with correct color
        hl.FillColor    = fillColor
        hl.OutlineColor = outlineColor

        local heartStr
        if hp <= 0 then
            heartStr = "💀 Dead"
        else
            heartStr = string.rep("♥", hearts) .. string.rep("♡", 4 - hearts)
        end

        local nameLabel = Instance.new("TextLabel")
        nameLabel.Size                   = UDim2.new(1, 0, 0.5, 0)
        nameLabel.Position               = UDim2.new(0, 0, 0, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text                   = player.Name .. "  " .. heartStr
        nameLabel.TextColor3             = hp <= 0 and Color3.new(1,1,1) or fillColor
        nameLabel.TextStrokeTransparency = 0
        nameLabel.TextStrokeColor3       = Color3.new(0, 0, 0)
        nameLabel.Font                   = Enum.Font.GothamBold
        nameLabel.TextScaled             = true
        nameLabel.Parent                 = frame

        local hpLabel = Instance.new("TextLabel")
        hpLabel.Size                   = UDim2.new(1, 0, 0.5, 0)
        hpLabel.Position               = UDim2.new(0, 0, 0.5, 0)
        hpLabel.BackgroundTransparency = 1
        hpLabel.Text                   = hp <= 0 and "Dead" or ("HP: " .. hp .. "  |  Hearts: " .. hearts .. "/4")
        hpLabel.TextColor3             = Color3.new(1, 1, 1)
        hpLabel.TextStrokeTransparency = 0
        hpLabel.TextStrokeColor3       = Color3.new(0, 0, 0)
        hpLabel.Font                   = Enum.Font.Gotham
        hpLabel.TextScaled             = true
        hpLabel.Parent                 = frame
    end

    table.insert(playerESPObjects, bb)
end
